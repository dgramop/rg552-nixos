# U-Boot commands to inspect Android boot configuration
# Run these at the U-Boot prompt to understand how Android boots

# === 1. Check current environment ===
echo "=== U-Boot Environment Variables ==="
printenv

# === 2. List partitions on eMMC (device 0) ===
echo "=== eMMC Partitions (device 0) ==="
mmc dev 0
mmc part

# === 3. Check what's in the boot partition ===
echo "=== Boot partition contents ==="
# Android typically uses partition 1 or 2 for boot
ls mmc 0:1
ls mmc 0:2

# === 4. Try to find Android boot files ===
echo "=== Looking for Android boot files ==="
# Common locations: /boot/, /Image, /zImage, /boot.img
ls mmc 0:1 /
ls mmc 0:1 /boot/
ls mmc 0:2 /
ls mmc 0:2 /boot/

# === 5. Check for Android boot script ===
echo "=== Looking for boot scripts ==="
ls mmc 0:1 /boot.scr
ls mmc 0:1 /boot/boot.scr
ls mmc 0:2 /boot.scr
ls mmc 0:2 /boot/boot.scr

# === 6. Check bootcmd (what U-Boot runs by default) ===
echo "=== Default boot command ==="
printenv bootcmd

# === 7. Look for device trees ===
echo "=== Looking for device tree files ==="
ls mmc 0:1 /dtb/
ls mmc 0:1 /boot/dtb/
ls mmc 0:2 /dtb/
ls mmc 0:2 /boot/dtb/
